76 void radix2(struct complex *x_signal, int N)
77 {
78  int n2, k1, N1, N2;
79  struct complex bfly[2];
80  N1=2;
81  N2=N/2;
82 
84  for (n2=0; n2<N2; n2++)
85  {
86  /* Don't hurt the butterfly */
87  bfly[0].r = (x_signal[n2].r + x_signal[N2 + n2].r);
88  bfly[0].i = (x_signal[n2].i + x_signal[N2 + n2].i);
89  bfly[1].r = (x_signal[n2].r - x_signal[N2 + n2].r) * MyCos(n2*2*M_PI/N) - ((x_signal[n2].i - x_signal[N2 + n2].i) * -1 * MySin(n2*2*M_PI/N));
90  bfly[1].i = (x_signal[n2].i - x_signal[N2 + n2].i) * MyCos(n2*2*M_PI/N) + ((x_signal[n2].r - x_signal[N2 + n2].r) * -1 * MySin(n2*2*M_PI/N));
91 
92  /* In-place results */
93  for (k1=0; k1<N1; k1++)
94  {
95  x_signal[n2 + N2*k1].r = bfly[k1].r;
96  x_signal[n2 + N2*k1].i = bfly[k1].i;
97  }
98  }
99  /* Don't recurse if we're down to one butterfly */
100  if (N2!=1)
101  for (k1=0; k1<N1; k1++)
102  {
103  radix2(&x_signal[N2*k1], N2);

115 {
116  int bits, i, j, k;
117  float tempr, tempi;
118 
119  for (i=0; i<MAXPOW; i++)
120  if (pow_2[i]==N) bits=i;

123  {
